1//! Sparkle - A Playwright implementation in Rust
2//!
3//! Sparkle provides a high-level API for browser automation, closely matching
4//! the Playwright Python API while being idiomatic to Rust.
5//!
6//! # Features
7//! - Full async/await support using Tokio
8//! - Type-safe builders for configuration
9//! - Comprehensive error handling with Result types
10//! - Support for Chromium (Chrome/Edge)
11//!
12//! # Example
13//! ```no_run
14//! use sparkle::async_api::Playwright;
15//!
16//! #[tokio::main]
17//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
18//!     let playwright = Playwright::new().await?;
19//!     let browser = playwright.chromium().launch(Default::default()).await?;
20//!     let page = browser.new_page().await?;
21//!     
22//!     page.goto("https://example.com", Default::default()).await?;
23//!     println!("Title: {}", page.title().await?);
24//!     
25//!     browser.close().await?;
26//!     Ok(())
27//! }
28//! ```

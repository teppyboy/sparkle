1//! Platform detection and configuration
2//!
3//! Detects the current operating system and architecture to determine
4//! which browser binaries to download.
5
6use std::env;
7
8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9pub enum Platform {
10    WindowsX64,
11    WindowsArm64,
12    LinuxX64,
13    LinuxArm64,
14    MacOsX64,
15    MacOsArm64,
16}
17
18impl Platform {
19    /// Detect the current platform
20    pub fn detect() -> anyhow::Result<Self> {
21        let os = env::consts::OS;
22        let arch = env::consts::ARCH;
23
24        match (os, arch) {
25            ("windows", "x86_64") => Ok(Platform::WindowsX64),
26            ("windows", "aarch64") => Ok(Platform::WindowsArm64),
27            ("linux", "x86_64") => Ok(Platform::LinuxX64),
28            ("linux", "aarch64") => Ok(Platform::LinuxArm64),
29            ("macos", "x86_64") => Ok(Platform::MacOsX64),
30            ("macos", "aarch64") => Ok(Platform::MacOsArm64),
31            _ => Err(anyhow::anyhow!("Unsupported platform: {} {}", os, arch)),
32        }
33    }
34
35    /// Get the Chrome download URL for this platform
36    pub fn chrome_download_url(&self, version: &str) -> String {
37        match self {
38            Platform::WindowsX64 => format!(
39                "https://storage.googleapis.com/chrome-for-testing-public/{}/win64/chrome-win64.zip",
40                version
41            ),
42            Platform::WindowsArm64 => format!(
43                "https://storage.googleapis.com/chrome-for-testing-public/{}/win-arm64/chrome-win-arm64.zip",
44                version
45            ),
46            Platform::LinuxX64 => format!(
47                "https://storage.googleapis.com/chrome-for-testing-public/{}/linux64/chrome-linux64.zip",
48                version
49            ),
50            Platform::LinuxArm64 => format!(
51                "https://storage.googleapis.com/chrome-for-testing-public/{}/linux-arm64/chrome-linux-arm64.zip",
52                version
53            ),
54            Platform::MacOsX64 => format!(
55                "https://storage.googleapis.com/chrome-for-testing-public/{}/mac-x64/chrome-mac-x64.zip",
56                version
57            ),
58            Platform::MacOsArm64 => format!(
59                "https://storage.googleapis.com/chrome-for-testing-public/{}/mac-arm64/chrome-mac-arm64.zip",
60                version
61            ),
62        }
63    }
64
65    /// Get the ChromeDriver download URL for this platform
66    pub fn chromedriver_download_url(&self, version: &str) -> String {
67        match self {
68            Platform::WindowsX64 => format!(
69                "https://storage.googleapis.com/chrome-for-testing-public/{}/win64/chromedriver-win64.zip",
70                version
71            ),
72            Platform::WindowsArm64 => format!(
73                "https://storage.googleapis.com/chrome-for-testing-public/{}/win-arm64/chromedriver-win-arm64.zip",
74                version
75            ),
76            Platform::LinuxX64 => format!(
77                "https://storage.googleapis.com/chrome-for-testing-public/{}/linux64/chromedriver-linux64.zip",
78                version
79            ),
80            Platform::LinuxArm64 => format!(
81                "https://storage.googleapis.com/chrome-for-testing-public/{}/linux-arm64/chromedriver-linux-arm64.zip",
82                version
83            ),
84            Platform::MacOsX64 => format!(
85                "https://storage.googleapis.com/chrome-for-testing-public/{}/mac-x64/chromedriver-mac-x64.zip",
86                version
87            ),
88            Platform::MacOsArm64 => format!(
89                "https://storage.googleapis.com/chrome-for-testing-public/{}/mac-arm64/chromedriver-mac-arm64.zip",
90                version
91            ),
92        }
93    }
94
95    /// Get the executable file extension for this platform
96    pub fn executable_extension(&self) -> &str {
97        match self {
98            Platform::WindowsX64 | Platform::WindowsArm64 => ".exe",
99            _ => "",
100        }
101    }
102
103    /// Get the platform name as a string
104    pub fn name(&self) -> &str {
105        match self {
106            Platform::WindowsX64 => "windows-x64",
107            Platform::WindowsArm64 => "windows-arm64",
108            Platform::LinuxX64 => "linux-x64",
109            Platform::LinuxArm64 => "linux-arm64",
110            Platform::MacOsX64 => "macos-x64",
111            Platform::MacOsArm64 => "macos-arm64",
112        }
113    }
114}
115
116impl std::fmt::Display for Platform {
117    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
118        write!(f, "{}", self.name())
119    }
120}

122#[cfg(test)]
123mod tests {
124    use super::*;
125
126    #[test]
127    fn test_platform_detection() {
128        let platform = Platform::detect();
129        assert!(platform.is_ok());
130    }
131
132    #[test]
133    fn test_chrome_url() {
134        let platform = Platform::WindowsX64;
135        let url = platform.chrome_download_url("120.0.6099.109");
136        assert!(url.contains("win64"));
137        assert!(url.contains("chrome"));
138    }
139
140    #[test]
141    fn test_chromedriver_url() {
142        let platform = Platform::LinuxX64;
143        let url = platform.chromedriver_download_url("120.0.6099.109");
144        assert!(url.contains("linux64"));
145        assert!(url.contains("chromedriver"));
146    }
